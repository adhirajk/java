Program to Copy Elements of One Array into Another Array

					
						

public class ArrayCopy {
    public static void main(String[] args) {
        int[] original = {1, 2, 3, 4, 5};
        int[] copy = new int[original.length];
        
        for (int i = 0; i < original.length; i++) {
            copy[i] = original[i];
        }

        System.out.print("Copied Array: ");
        for (int element : copy) {
            System.out.print(element + " ");
        }
    }
}





Outout:-   

Copied Array: 1 2 3 4 5

Program to Find the Frequency of Each Element in the Array

					
						

import java.util.HashMap;

public class FrequencyCount {
    public static void main(String[] args) {
        int[] array = {1, 2, 2, 3, 3, 3, 4};
        HashMap frequencyMap = new HashMap<>();
        
        for (int number : array) {
            frequencyMap.put(number, frequencyMap.getOrDefault(number, 0) + 1);
        }

        System.out.println("Frequency of each element: " + frequencyMap);
    }
}





Outout:-   

Frequency of each element: {1=1, 2=2, 3=3, 4=1}

Program to Left Rotate the Elements of an Array

					
						

public class LeftRotate {
    public static void main(String[] args) {
        int[] array = {1, 2, 3, 4, 5};
        int n = 1; // Number of positions to rotate
        leftRotate(array, n);

        System.out.print("Left Rotated Array: ");
        for (int element : array) {
            System.out.print(element + " ");
        }
    }

    public static void leftRotate(int[] arr, int d) {
        int n = arr.length;
        int[] temp = new int[d];
        for (int i = 0; i < d; i++) {
            temp[i] = arr[i];
        }
        for (int i = 0; i < n - d; i++) {
            arr[i] = arr[i + d];
        }
        for (int i = 0; i < d; i++) {
            arr[n - d + i] = temp[i];
        }
    }
}





Outout:-   

Left Rotated Array: 2 3 4 5 1

Program to Right Rotate the Elements of an Array

					
						

public class RightRotate {
    public static void main(String[] args) {
        int[] array = {1, 2, 3, 4, 5};
        int n = 1; // Number of positions to rotate
        rightRotate(array, n);

        System.out.print("Right Rotated Array: ");
        for (int element : array) {
            System.out.print(element + " ");
        }
    }

    public static void rightRotate(int[] arr, int d) {
        int n = arr.length;
        d = d % n; // To handle d >= n
        int[] temp = new int[d];
        for (int i = n - d; i < n; i++) {
            temp[i - (n - d)] = arr[i];
        }
        for (int i = n - 1; i >= d; i--) {
            arr[i] = arr[i - d];
        }
        for (int i = 0; i < d; i++) {
            arr[i] = temp[i];
        }
    }
}





Outout:-   

Right Rotated Array: 5 1 2 3 4

Program to Remove Duplicate Elements in an Array

					
						

import java.util.Arrays;

public class RemoveDuplicates {
    public static void main(String[] args) {
        int[] array = {5, 2, 9, 1, 2, 6, 9};
        Arrays.sort(array);
        System.out.print("Array without Duplicates: ");
        for (int i = 0; i < array.length; i++) {
            if (i == 0 || array[i] != array[i - 1]) {
                System.out.print(array[i] + " ");
            }
        }
    }
}





Outout:-   

Array without Duplicates: 1 2 5 6 9

Program to Print Odd and Even Numbers from an Array

					
						

public class OddEvenNumbers {
    public static void main(String[] args) {
        int[] array = {1, 2, 3, 4, 5};
        
        System.out.print("Even Numbers: ");
        for (int number : array) {
            if (number % 2 == 0) {
                System.out.print(number + " ");
            }
        }

        System.out.print("\nOdd Numbers: ");
        for (int number : array) {

                System.out.print(number + " ");
            }
        }
    }
}
