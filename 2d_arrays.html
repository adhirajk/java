Enter the Matrix and Print in Matrix Format



import java.util.Scanner;

public class MatrixInput {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Enter number of rows: ");
        int rows = scanner.nextInt();
        System.out.print("Enter number of columns: ");
        int cols = scanner.nextInt();
        
        int[][] matrix = new int[rows][cols];
        
        System.out.println("Enter the matrix elements:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                matrix[i][j] = scanner.nextInt();
            }
        }
        
        System.out.println("The entered matrix is:");
        for (int[] row : matrix) {
            for (int elem : row) {
                System.out.print(elem + " ");
            }
            System.out.println();
        }
    }
}

Multiply Two Matrices

                     

import java.util.Scanner;

public class MatrixMultiplication {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Enter number of rows of first matrix: ");
        int rows1 = scanner.nextInt();
        System.out.print("Enter number of columns of first matrix: ");
        int cols1 = scanner.nextInt();
        
        System.out.print("Enter number of columns of second matrix: ");
        int cols2 = scanner.nextInt();
        
        int[][] matrix1 = new int[rows1][cols1];
        int[][] matrix2 = new int[cols1][cols2];
        int[][] product = new int[rows1][cols2];
        
        System.out.println("Enter first matrix:");
        for (int i = 0; i < rows1; i++) {
            for (int j = 0; j < cols1; j++) {
                matrix1[i][j] = scanner.nextInt();
            }
        }
        
        System.out.println("Enter second matrix:");
        for (int i = 0; i < cols1; i++) {
            for (int j = 0; j < cols2; j++) {
                matrix2[i][j] = scanner.nextInt();
            }
        }
        
        for (int i = 0; i < rows1; i++) {
            for (int j = 0; j < cols2; j++) {
                for (int k = 0; k < cols1; k++) {
                    product[i][j] += matrix1[i][k] * matrix2[k][j];
                }
            }
        }
        
        System.out.println("Product of the matrices:");
        for (int[] row : product) {
            for (int elem : row) {
                System.out.print(elem + " ");
            }
            System.out.println();
        }
    }
}

Determine Whether Two Matrices Are Equal




import java.util.Scanner;

public class MatrixEquality {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        
        System.out.print("Enter number of rows: ");
        int rows = scanner.nextInt();
        System.out.print("Enter number of columns: ");
        int cols = scanner.nextInt();
        
        int[][] matrix1 = new int[rows][cols];
        int[][] matrix2 = new int[rows][cols];
        
        System.out.println("Enter first matrix:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                matrix1[i][j] = scanner.nextInt();
            }
        }
        
        System.out.println("Enter second matrix:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                matrix2[i][j] = scanner.nextInt();
            }
        }
        
        boolean isEqual = true;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (matrix1[i][j] != matrix2[i][j]) {
                    isEqual = false;
                    break;
                }
            }
        }
        
        if (isEqual) {
            System.out.println("The matrices are equal.");
        } else {
            System.out.println("The matrices are not equal.");
        }
    }
}

                matrix[i][j] = scanner.nextInt();
            }
        }
        
        System.out.println("Lower triangular matrix:");

                    System.out.print(matrix[i][j] + " ");

    }

Find the Frequency of Odd & Even Numbers in the Matrix

					
						

import java.util.Scanner;

public class OddEvenFrequency {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter number of rows: ");
        int rows = scanner.nextInt();
        System.out.print("Enter number of columns: ");
        int cols = scanner.nextInt();

        int[][] matrix = new int[rows][cols];
        int oddCount = 0, evenCount = 0;

        System.out.println("Enter the matrix elements:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                matrix[i][j] = scanner.nextInt();
                if (matrix[i][j] % 2 == 0) {
                    evenCount++;
                } else {
                    oddCount++;
                }
            }
        }

        System.out.println("Odd numbers frequency: " + oddCount);
        System.out.println("Even numbers frequency: " + evenCount);
    }
}

Enter number of columns: 3
Enter the matrix elements:
1 2 3
4 5 6
Odd numbers frequency: 3
Even numbers frequency: 3

Find the Sum of Each Row and Each Column of a Matrix

					
						

import java.util.Scanner;

public class RowColumnSum {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter number of rows: ");
        int rows = scanner.nextInt();
        System.out.print("Enter number of columns: ");
        int cols = scanner.nextInt();

        int[][] matrix = new int[rows][cols];
        int[] rowSum = new int[rows];
        int[] colSum = new int[cols];

        System.out.println("Enter the matrix elements:");
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                matrix[i][j] = scanner.nextInt();
                rowSum[i] += matrix[i][j];
                colSum[j] += matrix[i][j];
            }
        }

        System.out.println("Sum of each row:");
        for (int sum : rowSum) {
            System.out.print(sum + " ");
        }
        System.out.println();

        System.out.println("Sum of each column:");
        for (int sum : colSum) {
            System.out.print(sum + " ");
        }
    }
}

Determine Whether a Given Matrix is an Identity Matrix

					
						

import java.util.Scanner;

public class IdentityMatrixCheck {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter number of rows and columns: ");
        int n = scanner.nextInt();

        int[][] matrix = new int[n][n];
        boolean isIdentity = true;

        System.out.println("Enter the matrix elements:");
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                matrix[i][j] = scanner.nextInt();
            }
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j && matrix[i][j] != 1) {
                    isIdentity = false;
                } else if (i != j && matrix[i][j] != 0) {
                    isIdentity = false;
                }
            }
        }

        if (isIdentity) {
            System.out.println("The matrix is an identity matrix.");
        } else {
            System.out.println("The matrix is not an identity matrix.");
        }
    }
}





Outout:-   

Enter number of rows and columns: 3
Enter the matrix elements:
1 0 0
0 1 0
0 0 1
The matrix is an identity matrix.
